SELECT e.employee_name,
       d.department_name,
       AVG(s.salary) AS avg_salary
FROM employees e
JOIN departments d
  ON e.department_id = d.id
JOIN salaries s
  ON e.id = s.employee_id
WHERE s.year > 2020
GROUP BY e.employee_name, d.department_name
HAVING AVG(s.salary) > 75000
ORDER BY avg_salary DESC;
